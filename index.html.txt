<body>
  <h2>১.৫০% যোগ হয়ে মোট কত পে করতে হবে</h2>

  <label>এমাউন্ট লিখুন:</label>
  <input id="amount" type="number" placeholder="উদাহরণ: 1000" />

  <button onclick="calc()">Calculate</button>

  <div id="output" class="result"></div>

  <script>
    function calc() {
      let amount = parseFloat(document.getElementById('amount').value);
      if (isNaN(amount) || amount < 0) {
        document.getElementById('output').innerHTML = "সঠিক এমাউন্ট দিন";
        return;
      }

      let payable = amount * 1.015; // 1.50% যোগ করা
      payable = payable.toFixed(2);

      document.getElementById('output').innerHTML =
        "মোট পেমেন্ট দিতে হবে: " + payable + " টাকা";
    }
  </script>
